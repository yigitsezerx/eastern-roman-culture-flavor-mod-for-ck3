roman_culture_character.1 = { #for regional culture game rule
	type = character_event
	title = roman_culture_character.1.t
	desc = roman_culture_character.1.desc
	theme = realm

		character = root
		animation = admiration
	}

		name = roman_culture_character.1.a
		add_prestige = minor_prestige_gain

			base = 40
			ai_value_modifier = {
				ai_zeal = 1
				ai_rationality = -1
				ai_boldness = -1
				ai_honor = 1
				ai_energy = 1
			}
			modifier = {
				add = 25
				has_trait = zealous
			}
			modifier = { #to prevent appointed roman governors from converting
				add = 100
				culture = { has_cultural_pillar = language_latin }
			}
			modifier = {
				add = 40
				faith = {
					has_doctrine = tenet_communal_identity
				}
			}
		}
	}

	option = {
		name = roman_culture_character.1.b
		add_prestige = miniscule_prestige_gain
		change_influence = miniscule_influence_gain

		    limit = {
				OR = {
				    primary_title = title:k_hellas

						OR = {
							de_jure_liege = title:k_hellas
							de_jure_liege.de_jure_liege = title:k_hellas
						}
					}
				}
			}
			set_culture = culture:eastern_roman
			if = {
				limit = { any_spouse = { is_playable_character = no } }
				every_spouse = {
					limit = {
						is_playable_character = no
					}
					set_culture = culture:eastern_roman
				}
			}
			if = {
				limit = {
					any_close_family_member = {
						is_playable_character = no
						NOT = { is_spouse_of = root }
					}
				}
				every_close_family_member = {
					limit = {
								is_playable_character = no
						NOT = { is_spouse_of = root }
					}
					custom = all_unlanded_family_members
					set_culture = culture:eastern_roman
				}
			}
			if = {
				limit = {
					any_child = {
						NOT = { is_spouse_of = root }
					}
				}
				every_child = {
					limit = {
						NOT = { is_spouse_of = root }
					}
					custom = all_children_custom
					set_culture = culture:eastern_roman
				}
			}
		}
		else_if = { #needs to be this way to avoid half of greece becoming thraco-roman
		    limit = {
				OR = {
				    primary_title = title:d_philippopolis
					primary_title = {
						de_jure_liege = title:d_philippopolis
					}
					primary_title = title:d_turnovo
					primary_title = {
						de_jure_liege = title:d_turnovo
					}
					primary_title = title:d_dobrudja
					primary_title = {
						de_jure_liege = title:d_dobrudja
					}
					primary_title = title:d_bulgaria
					primary_title = {
						de_jure_liege = title:d_bulgaria
					}
					primary_title = title:d_vidin
					primary_title = {
						de_jure_liege = title:d_vidin
					}
					primary_title = title:k_thessalonika
				}
			}
			set_culture = culture:thraco_roman
			if = {
				limit = { any_spouse = { is_playable_character = no } }
				every_spouse = {
					limit = {
						is_playable_character = no
					}
					set_culture = culture:thraco_roman
				}
			}
			if = {
				limit = {
					any_close_family_member = {
						is_playable_character = no
						NOT = { is_spouse_of = root }
					}
				}
				every_close_family_member = {
					limit = {
								is_playable_character = no
						NOT = { is_spouse_of = root }
					}
					custom = all_unlanded_family_members
					set_culture = culture:thraco_roman
				}
			}
			if = {
				limit = {
					any_child = {
						NOT = { is_spouse_of = root }
					}
				}
				every_child = {
					limit = {
						NOT = { is_spouse_of = root }
					}
					custom = all_children_custom
					set_culture = culture:thraco_roman
				}
			}
		}
		else_if = { #needs to be this way to avoid half of greece becoming thraco-roman
		    limit = {
				OR = {
				    primary_title = title:d_thrace
					primary_title = {
						de_jure_liege = title:d_thrace
					}
					primary_title = title:d_strymon
					primary_title = {
						de_jure_liege = title:d_strymon
					}
				}
			}
			set_culture = eastern_roman
			if = {
				limit = { any_spouse = { is_playable_character = no } }
				every_spouse = {
					limit = {
						is_playable_character = no
					}
					set_culture = culture:eastern_roman
				}
			}
			if = {
				limit = {
					any_close_family_member = {
						is_playable_character = no
						NOT = { is_spouse_of = root }
					}
				}
				every_close_family_member = {
					limit = {
								is_playable_character = no
						NOT = { is_spouse_of = root }
					}
					custom = all_unlanded_family_members
					set_culture = culture:eastern_roman
				}
			}
			if = {
				limit = {
					any_child = {
						NOT = { is_spouse_of = root }
					}
				}
				every_child = {
					limit = {
						NOT = { is_spouse_of = root }
					}
					custom = all_children_custom
					set_culture = culture:eastern_roman
				}
			}
		}
		else_if = {
		    limit = {
				OR = {
				    primary_title = title:k_syria
					primary_title = {
						OR = {
							de_jure_liege = title:k_syria
							de_jure_liege.de_jure_liege = title:k_syria
						}
					}
				}
			}
			set_culture = culture:levantine_roman
			if = {
				limit = { any_spouse = { is_playable_character = no } }
				every_spouse = {
					limit = {
						is_playable_character = no
					}
					set_culture = culture:levantine_roman
				}
			}
			if = {
				limit = {
					any_close_family_member = {
						is_playable_character = no
						NOT = { is_spouse_of = root }
					}
				}
				every_close_family_member = {
					limit = {

					}
					custom = all_unlanded_family_members
					set_culture = culture:levantine_roman
				}
			}
			if = {
				limit = {
					any_child = {
						NOT = { is_spouse_of = root }
					}
				}
				every_child = {
					limit = {
						NOT = { is_spouse_of = root }
					}
					custom = all_children_custom
					set_culture = culture:levantine_roman
				}
			}
		}
		else_if = {
		    limit = {
				OR = {
				    primary_title = title:k_egypt
					primary_title = {
						OR = {
							de_jure_liege = title:k_egypt
							de_jure_liege.de_jure_liege = title:k_egypt
						}
					}
				}
			}
			set_culture = culture:egypto_roman
			if = {

					custom = all_unlanded_family_members
					set_culture = culture:egypto_roman
				}
			}
			if = {
				limit = {
					any_child = {
						NOT = { is_spouse_of = root }
					}
				}
				every_child = {
					limit = {
						NOT = { is_spouse_of = root }
					}
					custom = all_children_custom
					set_culture = culture:egypto_roman
				}
			}
		}
		else = { #fallback and temporarily for illyria
			set_culture = culture:eastern_roman
			if = {
				limit = { any_spouse = { is_playable_character = no } }
				every_spouse = {
					limit = {
						is_playable_character = no
					}
					set_culture = culture:eastern_roman
				}
			}
			if = {
				limit = {
					any_close_family_member = {
						is_playable_character = no
						NOT = { is_spouse_of = root }
					}
				}
				every_close_family_member = {
					limit = {
								is_playable_character = no
						NOT = { is_spouse_of = root }
					}
					custom = all_unlanded_family_members
					set_culture = culture:eastern_roman
				}
			}
			if = {
				limit = {
					any_child = {
